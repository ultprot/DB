char* random_string(int size) {
	char *strChar = new char[size];
	for (int i = 0; i < size; i++) {
		strChar[i] = 97 + rand() % 25;
	}
	strChar[size] = 0;
	return strChar;
}
char* random_string_number(int size) {
	char *strChar = new char[size];
	for (int i = 0; i < size; i++) {
		strChar[i] = 48 + rand() % 10;
	}
	strChar[size] = 0;
	return strChar;
}
